async function fetchIds(baseUrl, ids) {
  const results = {};
  await Promise.all(ids.map(async id => {
    const url = `${baseUrl.replace(/\/$/, "")}/${encodeURIComponent(id)}`;
    try {
      const res = await fetch(url, {
        credentials: "omit"
      });
      if (!res.ok) {
        results[id] = {
          error: `HTTP ${res.status}`
        };
        return;
      }
      const html = await res.text();
      const doc = new DOMParser().parseFromString(html, "text/html");
      const titleEl = doc.querySelector("title");
      const canonicalEl = doc.querySelector("link[rel='canonical']");
      let lastSegment = null;
      if (canonicalEl) {
        const href = canonicalEl.href || canonicalEl.getAttribute("href") || "";
        const segments = href.replace(/\/+$/, "").split("/");
        const last = segments[segments.length - 1] || "";
        lastSegment = /^\d+$/.test(last) ? last : null;
      }
      results[id] = {
        title: titleEl ? titleEl.textContent.trim() : null,
        id: lastSegment
      };
    } catch (err) {
      results[id] = {
        error: err.message
      };
    }
  }));
  return results;
}


function expandList(str) {
  return str
    .split(/\s*,\s*/g)
    .flatMap(token => {
      if (!token) return [];
      const range = token.match(/^(\d+)\s*-\s*(\d+)$/);
      if (range) {
        const a = Number(range[1]),
          b = Number(range[2]);
        if (a <= b) return Array.from({
          length: b - a + 1
        }, (_, i) => String(a + i));
        return Array.from({
          length: a - b + 1
        }, (_, i) => String(a - i));
      }
      return [token];
    })
}

(async () => {
  const base = "https://support.apple.com/en-us";
  const raw = prompt("Enter support IDs separated by commas, bruh\n also fam ranges allowed (e.g. 3, 5-8, 67, 420):");

  const ids = expandList(raw);
  const output = await fetchIds(base, ids);
  console.log(Object.values(output).map(entry => `${entry.id}\t${entry.title.replace(" - Apple Support", "")}\t`).join("\n"))
})();
